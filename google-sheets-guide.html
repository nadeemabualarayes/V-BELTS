// ═══════════════════════════════════════════
// نظام قطع الغيار — Google Apps Script API
// انسخ هذا الكود كاملاً في Apps Script
// ═══════════════════════════════════════════

const SHEET_NAME = 'parts';

// نقطة الدخول الرئيسية — تعالج كل الطلبات
function doGet(e) {
  const action = e.parameter.action;
  let result;

  if (action === 'getAll') {
    result = getAllParts();
  } else if (action === 'search') {
    result = searchPart(e.parameter.q);
  } else {
    result = { error: 'action غير معروف' };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// يعالج الإضافة والتعديل والحذف
function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const action = data.action;
  let result;

  if (action === 'add') {
    result = addPart(data.part);
  } else if (action === 'update') {
    result = updatePart(data.part);
  } else if (action === 'delete') {
    result = deletePart(data.id);
  } else {
    result = { error: 'action غير معروف' };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// جلب كل القطع
function getAllParts() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  const headers = rows[0];
  const parts = [];

  for (let i = 1; i < rows.length; i++) {
    if (!rows[i][0]) continue; // تخطي الصفوف الفارغة
    const part = {};
    headers.forEach((h, j) => {
      let val = rows[i][j];
      // تحويل JSON strings إلى objects
      if ((h === 'dims' || h === 'fitment') && typeof val === 'string' && val) {
        try { val = JSON.parse(val); } catch(e) { val = h === 'fitment' ? [] : {}; }
      }
      part[h] = val;
    });
    parts.push(part);
  }
  return { success: true, data: parts };
}

// إضافة قطعة جديدة
function addPart(part) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];

  // توليد ID جديد
  part.id = Date.now();

  const row = headers.map(h => {
    let val = part[h] ?? '';
    // تحويل objects إلى JSON strings للحفظ
    if (typeof val === 'object' && val !== null) val = JSON.stringify(val);
    return val;
  });

  sheet.appendRow(row);
  return { success: true, id: part.id, message: 'تمت الإضافة' };
}

// تعديل قطعة موجودة
function updatePart(part) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(part.id)) {
      const row = headers.map(h => {
        let val = part[h] ?? '';
        if (typeof val === 'object' && val !== null) val = JSON.stringify(val);
        return val;
      });
      sheet.getRange(i + 1, 1, 1, row.length).setValues([row]);
      return { success: true, message: 'تم التحديث' };
    }
  }
  return { error: 'القطعة غير موجودة' };
}

// حذف قطعة
function deletePart(id) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(id)) {
      sheet.deleteRow(i + 1);
      return { success: true, message: 'تم الحذف' };
    }
  }
  return { error: 'القطعة غير موجودة' };
}

// بحث عن قطعة
function searchPart(q) {
  const all = getAllParts();
  if (!all.success) return all;
  const ql = q.toLowerCase();
  const results = all.data.filter(p =>
    String(p.num).toLowerCase().includes(ql) ||
    String(p.name).includes(q) ||
    (p.dims && p.dims.altcode && String(p.dims.altcode).toLowerCase().includes(ql))
  );
  return { success: true, data: results };
}
